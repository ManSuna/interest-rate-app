useEffect(() => {
  const ids = [123, 456];

  const fetchUsers = async () => {
    const results = await Promise.all(
      ids.map((id) => fetch(`/api/user/${id}`).then((res) => res.json()))
    );

    results.forEach((userData, index) => {
      console.log(`User ${ids[index]}:`, userData);
    });
  };

  fetchUsers();
}, []);
