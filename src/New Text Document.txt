import React, { useState } from 'react';
import axios from 'axios';
import './App.css';

function App() {
  const [formData, setFormData] = useState({
    businessDate: '',
    closingTime: '',
    amount: '',
    rate: '',
  });

  const [jwtResponse, setJwtResponse] = useState(null);
  const [error, setError] = useState('');

  const handleChange = (e) => {
    setFormData({ ...formData, [e.target.name]: e.target.value });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError('');
    try {
      const response = await axios.post('http://localhost:5000/api/submit', formData);
      setJwtResponse(response.data.token); // Assuming backend returns { token: "JWT..." }
    } catch (err) {
      setError('Failed to submit data. Check console for details.');
      console.error(err);
    }
  };

  return (
    <div className="App">
      <h2>Send Business Data</h2>
      <form onSubmit={handleSubmit}>
        <input type="date" name="businessDate" onChange={handleChange} required />
        <input type="time" name="closingTime" onChange={handleChange} required />
        <input type="number" name="amount" placeholder="Amount" onChange={handleChange} required />
        <input type="number" step="0.01" name="rate" placeholder="Rate" onChange={handleChange} required />
        <button type="submit">Submit</button>
      </form>

      {jwtResponse && (
        <div>
          <h3>JWT Response:</h3>
          <textarea readOnly rows="6" value={jwtResponse} style={{ width: '100%' }} />
        </div>
      )}

      {error && <p style={{ color: 'red' }}>{error}</p>}
    </div>
  );
}

export default App;
